```js
let scrollTop = document.documentElement.scrollTop||document.body.scrollTop;
let scrollLeft = document.documentElment.scrollLeft||document.body.scrollLeft;
```

#### 阻止事件冒泡

```js
if(e.stopPropagation){
  e.stopPropagation();
}else{
  e.cancelBubble = true;
}
```

#### 阻止默认事件

```js
if(e.preventDefault){
 	e.preventDefault();
}else if(e.returnValue== undefined){
  e.returnValue = false;
}else{
  return false;
}
```
